from pymongo import MongoClient
from datetime import datetime, timezone
client = MongoClient('mongodb://localhost:27017')
db = client['pet-os-live-test_database']
now = datetime.now(timezone.utc).isoformat()

for bp in db.breed_products.find({'is_active': True}, {'_id': 0}):
    pid = bp.get('id')
    if not pid: continue
    pm = db.products_master.find_one({'id': pid}, {'_id': 0, 'product_type': 1, 'visibility': 1})
    if not pm: continue
    correct_type = bp.get('product_type') or bp.get('category') or 'soul_made'
    img = bp.get('cloudinary_url') or bp.get('mockup_url')
    update = {'product_type': correct_type, 'soul_made': True, 'updated_at': now}
    if img: update.update({'cloudinary_url': img, 'image_url': img, 'image': img})
    if not pm.get('visibility'): update['visibility'] = {'status': 'active'}
    db.products_master.update_one({'id': pid}, {'$set': update})
print('Sync complete')

import pymongo
from collections import defaultdict
client = pymongo.MongoClient('mongodb://localhost:27017', serverSelectionTimeoutMS=3000)
db = client['pet-os-live-test_database']

PILLARS = ["dine","care","play","go","learn","celebrate","shop","paperwork","emergency","farewell","adopt"]
breeds = sorted(db.breed_products.distinct("breed"))
visible = list(db.products_master.find(
    {"is_active": {"$ne": False}, "visibility.status": {"$nin": ["inactive","archived"]}},
    {"_id": 0, "id": 1, "name": 1, "pillar": 1, "breed_tags": 1, "breed": 1}
))
breed_set = set(breeds)

def get_product_breeds(doc):
    bt = doc.get("breed_tags") or []
    br = doc.get("breed") or ""
    pid = (doc.get("id") or "").lower()
    name = (doc.get("name") or "").lower()
    matched = set()
    if bt and isinstance(bt, list):
        matched |= {b for b in bt if b in breed_set}
    if br and br in breed_set:
        matched.add(br)
    for b in breed_set:
        for bv in [b, b.replace("_"," "), b.replace("_","-")]:
            if bv in pid or bv in name:
                matched.add(b)
                break
    return matched

vis_counts = defaultdict(lambda: defaultdict(int))
for doc in visible:
    p = doc.get("pillar","")
    if p not in PILLARS: continue
    for b in get_product_breeds(doc):
        vis_counts[b][p] += 1

src_agg = list(db.breed_products.aggregate([{"$group":{"_id":{"breed":"$breed","pillar":"$pillar"},"count":{"$sum":1}}}]))
src_counts = defaultdict(lambda: defaultdict(int))
for row in src_agg:
    src_counts[row["_id"]["breed"]][row["_id"]["pillar"]] = row["count"]

gaps = []
for breed in breeds:
    for p in PILLARS:
        v = vis_counts[breed][p]
        s = src_counts[breed][p]
        if v < s:
            gaps.append((breed, p, v, s))

print(f"Total gaps found: {len(gaps)}")
for breed, p, v, s in gaps:
    print(f"  {breed} | {p}: visible={v} source={s} (missing {s-v})")
